                            scale: 1, 
                            stagger: 0.2,

                        { opacity: 1, y: 0, scale: 1, duration: 1, ease: 'back.out(1.7)' },

                    );

                        { opacity: 0, y: 20 },
                        { opacity: 1, y: 0, duration: 0.8 },

                    );

                if (subtitle) {
                    tl.fromTo(subtitle,
                        { opacity: 0, y: 20 },
                        { opacity: 1, y: 0, duration: 0.8 },
                        '-=0.6'
                    );
                }

                // Animation ‡∏™‡∏≥‡∏´‡∏£‡∏±‡∏ö footer
